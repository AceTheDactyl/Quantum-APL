Python Usage

from quantum_apl_python import (
    GOLDEN_SAMPLE,
    get_golden_sample_bytes,
    verify_golden_sample,
    verify_golden_sample_detailed,

    extract_and_verify_golden_sample,
)

# Get the canonical signature

# Detailed verification with diagnostics

print(f"Valid: {result.is_valid}")
print(f"Matching thresholds: {result.matching_count}/9")

# Embed in image header (MRP-LSB)
image_with_header = embed_golden_sample_header(image_array)

# Extract and verify from image
verification = extract_and_verify_golden_sample(image_array)
print(f"Signature present: {verification.is_valid}")
